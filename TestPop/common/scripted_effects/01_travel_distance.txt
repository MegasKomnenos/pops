# start should always be a land province
travel_distance_dijkstra = {	
	$start$ = {
		set_variable = { name = travel_distance_distance value = 0.001 }
		set_variable = { name = travel_distance_prev value = this }
		
		add_to_global_variable_list = { name = travel_distance_fray target = this }
		add_to_global_variable_list = { name = travel_distance_visited target = this }
	}
	
	[[start_list]
		if = {
			limit = {
				or = {
					always = no
					
					[[start_list_global]
						always = $start_list_global$
					]
				}
			}
			every_in_global_list = {
				variable = $start_list$
				
				set_variable = { name = travel_distance_distance value = 0.001 }
				set_variable = { name = travel_distance_prev value = this }
				
				add_to_global_variable_list = { name = travel_distance_fray target = this }
				add_to_global_variable_list = { name = travel_distance_visited target = this }
			}
		}
		else = {
			every_in_list = {
				variable = $start_list$
				
				set_variable = { name = travel_distance_distance value = 0.001 }
				set_variable = { name = travel_distance_prev value = this }
				
				add_to_global_variable_list = { name = travel_distance_fray target = this }
				add_to_global_variable_list = { name = travel_distance_visited target = this }
			}
		}
	]
	
	while = {
		limit = {
			has_global_variable_list = travel_distance_fray
			
			global_variable_list_size = { name = travel_distance_fray value >= 1 }
		}
		count = 100000
		
		ordered_in_global_list = {
			variable = travel_distance_fray
			order_by = travel_distance_order_value
			position = 0
			
			set_global_variable = { name = travel_distance_select value = this }
			
			remove_list_global_variable = { name = travel_distance_fray target = this }
		}
		
		global_var:travel_distance_select = {
			if = {
				limit = {
					has_variable_list = prov_neighb
				}
				every_in_list = {
					variable = prov_neighb
					limit = {
						has_variable = travel_distance_distance
						
						var:travel_distance_distance = {
							compare_value > global_var:travel_distance_select.var:travel_distance_distance
						}
					}
					get_distance_neighb = { 
						one = global_var:travel_distance_select
						other = this
						base = global_var:travel_distance_select.var:travel_distance_distance 
						return = travel_distance_distance_new
					}
					
					if = {
						limit = {
							var:travel_distance_distance_new <= $max$
							
							var:travel_distance_distance_new = {
								compare_value < prev.var:travel_distance_distance
							}
						}
						add_to_global_variable_list = { name = travel_distance_fray target = this }
						add_to_global_variable_list = { name = travel_distance_visited target = this }
						
						set_variable = { name = travel_distance_distance value = var:travel_distance_distance_new }
						set_variable = { name = travel_distance_prev value = global_var:travel_distance_select }
					}
				}
			}
			if = {
				limit = {
					has_variable_list = prov_port
				}
				every_in_list = {
					variable = prov_port
					limit = {
						has_variable = travel_distance_distance
						
						var:travel_distance_distance = {
							compare_value > global_var:travel_distance_select.var:travel_distance_distance
						}
					}
					get_distance_port = { 
						one = global_var:travel_distance_select
						other = this
						base = global_var:travel_distance_select.var:travel_distance_distance 
						return = travel_distance_distance_new
					}
					
					if = {
						limit = {
							var:travel_distance_distance_new <= $max$
							
							var:travel_distance_distance_new = {
								compare_value < prev.var:travel_distance_distance
							}
						}
						add_to_global_variable_list = { name = travel_distance_fray target = this }
						add_to_global_variable_list = { name = travel_distance_visited target = this }
						
						set_variable = { name = travel_distance_distance value = var:travel_distance_distance_new }
						set_variable = { name = travel_distance_prev value = global_var:travel_distance_select }
					}
				}
			}
		}
	}
	
	every_in_global_list = {
		variable = travel_distance_visited
		limit = {
			var:prov_sea = 2
		}
		while = {
			limit = {
				var:travel_distance_prev.var:prov_sea = 2
			}
			set_variable = { name = travel_distance_prev value = var:travel_distance_prev.var:travel_distance_prev }
		}
	}
	every_in_global_list = {
		variable = travel_distance_visited
		limit = {
			var:prov_sea = 1
		}
		while = {
			limit = {
				var:travel_distance_prev.var:prov_sea = 2
			}
			set_variable = { name = travel_distance_prev value = var:travel_distance_prev.var:travel_distance_prev }
		}
	}
	
	every_in_global_list = {
		variable = travel_distance_visited
		
		set_variable = { name = $return_dist$ value = var:travel_distance_distance }
		set_variable = { name = $return_prev$ value = var:travel_distance_prev }
		
		remove_variable = travel_distance_distance_new
		remove_variable = travel_distance_distance
		remove_variable = travel_distance_prev
	}
	
	clear_global_variable_list = travel_distance_fray
	clear_global_variable_list = travel_distance_visited
	remove_global_variable = travel_distance_select
}

# start should always be a land province
find_path_dijkstra = {	
	$start$ = {
		set_variable = { name = find_path_distance value = 0 }
		set_variable = { name = find_path_prev value = this }
		
		add_to_global_variable_list = { name = find_path_fray target = this }
		add_to_global_variable_list = { name = find_path_visited target = this }
	}
	$dest$ = {
		set_variable = { name = find_path_dest }
		set_variable = { name = find_path_distance value = 210000 }
		set_variable = { name = find_path_prev value = this }
	}
	
	while = {
		limit = {
			has_global_variable_list = find_path_fray
			
			global_variable_list_size = { name = find_path_fray value >= 1 }
		}
		count = 100000
		
		ordered_in_global_list = {
			variable = find_path_fray
			order_by = find_path_order_value
			position = 0
			
			set_global_variable = { name = find_path_select value = this }
			
			remove_list_global_variable = { name = find_path_fray target = this }
		}
		
		global_var:find_path_select = {
			if = {
				limit = {
					has_variable = find_path_dest
				}
				set_global_variable = { name = find_path_dest_found }
				
				clear_global_variable_list = find_path_fray
			}
			else = {
				if = {
					limit = {
						has_variable_list = prov_neighb
					}
					every_in_list = {
						variable = prov_neighb
						limit = {
							has_variable = find_path_distance
						}
						get_distance_neighb = { 
							one = global_var:find_path_select
							other = this
							base = global_var:find_path_select.var:find_path_distance 
							return = find_path_distance_new
						}
						
						if = {
							limit = {
								var:find_path_distance_new = {
									compare_value < prev.var:find_path_distance
								}
							}
							add_to_global_variable_list = { name = find_path_fray target = this }
							add_to_global_variable_list = { name = find_path_visited target = this }
							
							set_variable = { name = find_path_distance value = var:find_path_distance_new }
							set_variable = { name = find_path_prev value = global_var:find_path_select }
						}
					}
				}
				if = {
					limit = {
						has_variable_list = prov_port
					}
					every_in_list = {
						variable = prov_port
						limit = {
							has_variable = find_path_distance
						}
						get_distance_port = { 
							one = global_var:find_path_select
							other = this
							base = global_var:find_path_select.var:find_path_distance 
							return = find_path_distance_new
						}
						
						if = {
							limit = {
								var:find_path_distance_new = {
									compare_value < prev.var:find_path_distance
								}
							}
							add_to_global_variable_list = { name = find_path_fray target = this }
							add_to_global_variable_list = { name = find_path_visited target = this }
							
							set_variable = { name = find_path_distance value = var:find_path_distance_new }
							set_variable = { name = find_path_prev value = global_var:find_path_select }
						}
					}
				}
			}
		}
	}
	
	every_in_global_list = {
		variable = find_path_visited
		limit = {
			var:prov_sea = 2
		}
		while = {
			limit = {
				var:find_path_prev.var:prov_sea = 2
			}
			set_variable = { name = find_path_prev value = var:find_path_prev.var:find_path_prev }
		}
	}
	every_in_global_list = {
		variable = find_path_visited
		limit = {
			var:prov_sea = 1
		}
		while = {
			limit = {
				var:find_path_prev.var:prov_sea = 2
			}
			set_variable = { name = find_path_prev value = var:find_path_prev.var:find_path_prev }
		}
	}
	
	clear_variable_list = $return_path$
	
	set_variable = { name = $return_dist$ value = -1 }
	
	if = {
		limit = {
			has_global_variable = find_path_dest_found
		}
		remove_global_variable = find_path_dest_found
		
		set_global_variable = { name = find_path_select value = $dest$ }
		
		while = {
			limit = {
				$start$.var:prov_id = {
					compare_value != global_var:find_path_select.var:prov_id
				}
			}
			add_to_variable_list = { name = $return_path$ target = global_var:find_path_select }
			
			set_global_variable = { name = find_path_select value = global_var:find_path_select.var:find_path_prev }
		}
		
		add_to_variable_list = { name = $return_path$ target = $start$ }
		
		set_variable = { name = $return_dist$ value = $dest$.var:find_path_distance }
	}
	
	$dest$ = {
		remove_variable = find_path_dest
		remove_variable = find_path_distance_new
		remove_variable = find_path_distance
		remove_variable = find_path_prev
	}
	
	every_in_global_list = {
		variable = find_path_visited
		
		remove_variable = find_path_distance_new
		remove_variable = find_path_distance
		remove_variable = find_path_prev
	}
	
	clear_global_variable_list = find_path_fray
	clear_global_variable_list = find_path_visited
	remove_global_variable = find_path_select
}

get_distance_neighb = {
	set_variable = { name = get_distance_neighb_x_diff value = $one$.var:prov_x }
	set_variable = { name = get_distance_neighb_y_diff value = $one$.var:prov_y }
	
	change_variable = { name = get_distance_neighb_x_diff subtract = $other$.var:prov_x }
	change_variable = { name = get_distance_neighb_y_diff subtract = $other$.var:prov_y }
	
	change_variable = { name = get_distance_neighb_x_diff multiply = var:get_distance_neighb_x_diff }
	change_variable = { name = get_distance_neighb_y_diff multiply = var:get_distance_neighb_y_diff }
	
	set_variable = { name = get_distance_neighb_diff value = var:get_distance_neighb_x_diff }
	change_variable = { name = get_distance_neighb_diff add = var:get_distance_neighb_y_diff }
	
	sqrt_effect = { inp = var:get_distance_neighb_diff return = get_distance_neighb_diff }
	
	set_variable = { name = get_distance_neighb_cost value = $one$.var:prov_cost }
	change_variable = { name = get_distance_neighb_cost add = $other$.var:prov_cost }
	
	set_variable = { name = $return$ value = var:get_distance_neighb_diff }
	change_variable = { name = $return$ multiply = var:get_distance_neighb_cost }
	change_variable = { name = $return$ add = $base$ }
	
	remove_variable = get_distance_neighb_x_diff
	remove_variable = get_distance_neighb_y_diff
	remove_variable = get_distance_neighb_diff
	remove_variable = get_distance_neighb_cost
}

get_distance_port = {
	set_variable = { name = get_distance_port_cost value = 100 }
	
	get_distance_neighb = {
		one = $one$
		other = $other$
		base = $base$
		return = $return$
	}
	
	change_variable = { name = $return$ add = var:get_distance_port_cost }
	
	remove_variable = get_distance_port_cost
}
