init_main = {
	init_prov_size = yes
	init_prov_data = yes
	
	every_province = {
		get_land_cost = { return = prov_cost }
	}
	every_in_global_list = {
		variable = every_water
		
		get_land_cost = { return = prov_cost }
	}
	
	init_pop_weight = yes
	init_pop_total = yes
	init_farm_score = yes
	init_pasture_score = yes
	init_forest_score = yes
	init_land = yes
	init_price = yes
	init_pay = yes
	
	modi_refresh = yes
	
	every_county = {
		set_variable = { name = pop_total value = 1 }
		set_variable = { name = farm_total value = 1 }
		set_variable = { name = pasture_total value = 1 }
		set_variable = { name = forest_total value = 1 }
		
		every_county_province = {
			if = {
				limit = {
					is_city = yes
				}
				set_variable = { name = pop_total_send value = 0.5 }
				set_variable = { name = farm_total_send value = 0.5 }
				set_variable = { name = pasture_total_send value = 0.5 }
				set_variable = { name = forest_total_send value = 0.5 }
			}
			else = {
				set_variable = { name = pop_total_send value = 1 }
				set_variable = { name = farm_total_send value = 1 }
				set_variable = { name = pasture_total_send value = 1 }
				set_variable = { name = forest_total_send value = 1 }
			}
			
			change_variable = { name = pop_total_send multiply = var:pop_total }
			change_variable = { name = farm_total_send multiply = var:farm_total }
			change_variable = { name = pasture_total_send multiply = var:pasture_total }
			change_variable = { name = forest_total_send multiply = var:forest_total }
			
			change_variable = { name = pop_total_send divide = 1000 }
			change_variable = { name = farm_total_send divide = 1000 }
			change_variable = { name = pasture_total_send divide = 1000 }
			change_variable = { name = forest_total_send divide = 1000 }
			
			change_variable = { name = pop_total_send multiply = 1000 }
			change_variable = { name = farm_total_send multiply = 1000 }
			change_variable = { name = pasture_total_send multiply = 1000 }
			change_variable = { name = forest_total_send multiply = 1000 }
			
			prev = {
				change_variable = { name = pop_total add = prev.var:pop_total_send }
				change_variable = { name = farm_total add = prev.var:farm_total_send }
				change_variable = { name = pasture_total add = prev.var:pasture_total_send }
				change_variable = { name = forest_total add = prev.var:forest_total_send }
			}
			
			change_variable = { name = pop_total subtract = var:pop_total_send }
			change_variable = { name = farm_total subtract = var:farm_total_send }
			change_variable = { name = pasture_total subtract = var:pasture_total_send }
			change_variable = { name = forest_total subtract = var:forest_total_send }
			
			remove_variable = pop_total_send
			remove_variable = pop_weight
			remove_variable = farm_total_send
			remove_variable = farm_score
			remove_variable = pasture_total_send
			remove_variable = pasture_score
			remove_variable = forest_total_send
			remove_variable = forest_score
		}
		
		set_variable = { name = pop_wealth value = var:pop_total }
		change_variable = { name = pop_wealth multiply = 4 }
	}
}
