setup_rulers = {
	every_ruler = {
		limit = {
			is_character = yes
			is_landed = yes
		}
		set_variable = { name = is_character value = 1 }
		
		if = {
			limit = {
				has_variable = trade_gold
				
				var:trade_gold > 0
			}
			remove_short_term_gold = 100000
			
			eff_binary_1_16 = { name = trade_gold eff = add_gold }
		}
		
		set_variable = { name = mil_maa value = var:trade_maa }
		set_variable = { name = mil_maintain value = var:trade_maintain }
		
		remove_variable = trade_gold
		remove_variable = mil_maa
		remove_variable = mil_maintain
	}
	
	modi_refresh = yes
	
	set_global_variable = { name = sim_i value = 1 }
	
	mil_main = yes
	stockpile_ai = yes
	task_ai = yes
	
	remove_global_variable = sim_i
}
