region_new_region = {
	create_character = {
		save_temporary_scope_as = region_new_region_tmp
		gender = male
		trait = character_not_1
		employer = global_var:arhat
		faith = global_var:arhat.faith
		culture = global_var:arhat.culture
		dynasty = none
	}
	
	add_to_global_variable_list = { name = region_regions target = scope:region_new_region_tmp }
	
	scope:region_new_region_tmp = {
		[[d0] set_variable = { name = region_name value = title:$d0$ } ]
		
		^^range20^
			[[d&range20&] 
				add_to_variable_list = { name = region_duchies target = title:$d&range20&$ }
				
				title:$d&range20&$ = {
					set_variable = { name = region_region value = prev }
				}
			]  
		^
		
		[[k0]
			set_variable = { name = region_name value = $k0$ }
			
			$k0$ = { 
				every_in_de_jure_hierarchy = { 
					limit = { 
						tier = tier_duchy 
						
						NOT = {
							has_variable = region_region
						}
						
						any_in_de_jure_hierarchy = {
							tier = tier_county
						}
					} 
					scope:region_new_region_tmp = { 
						add_to_variable_list = { name = region_duchies target = prev } 
						
						prev = {
							set_variable = { name = region_region value = prev }
						}
					} 
				} 
			}
		]
		
		set_variable = { name = region_x value = 0 }
		set_variable = { name = region_y value = 0 }
		set_variable = { name = region_new_region_tmp value = 0 }
		
		every_in_list = {
			variable = region_duchies
			
			every_in_de_jure_hierarchy = {
				limit = {
					tier = tier_barony
				}
				title_province = {
					scope:region_new_region_tmp = {
						change_variable = { name = region_x add = { value = prev.var:prov_x multiply = prev.var:prov_size divide = 100 } }
						change_variable = { name = region_y add = { value = prev.var:prov_y multiply = prev.var:prov_size divide = 100 } }
						change_variable = { name = region_new_region_tmp add = { value = prev.var:prov_size divide = 100 } }
					}
				}
			}
		}
		
		change_variable = { name = region_x divide = var:region_new_region_tmp }
		change_variable = { name = region_y divide = var:region_new_region_tmp }
		
		remove_variable = region_new_region_tmp
	}
}
