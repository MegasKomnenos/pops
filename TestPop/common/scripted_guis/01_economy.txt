toggle_county_trade = { 
	scope = province
	is_shown = { 
		always = yes
	}
	is_valid = { 
		always = yes
	}
	effect = { 
		if = { 
			limit = { 
				has_variable = toggled_county_trade
			}
			remove_variable = toggled_county_trade
			remove_variable = info_trade_county_price_food

			remove_variable = info_trade_county_price_salted

			remove_variable = info_trade_county_price_exotics_w
			remove_variable = info_trade_county_price_exotics_e
			remove_variable = info_trade_county_price_exotics_g
			remove_variable = info_trade_county_price_textile
			remove_variable = info_trade_county_price_pottery
			remove_variable = info_trade_county_price_tools
		}
		else = { 
			set_variable = { 
				name = toggled_county_trade
				value = 1
			}
			set_variable = { 
				name = info_trade_county_price_food
				value = county.var:trade_price_food
			}
			set_variable = { 
				name = info_trade_county_price_cereal
				value = county.var:trade_price_cereal
			}
			set_variable = { 
				name = info_trade_county_price_salted
				value = county.var:trade_price_salted
			}
			set_variable = { 
				name = info_trade_county_price_salt
				value = county.var:trade_price_salt
			}
			set_variable = { 
				name = info_trade_county_price_mud
				value = county.var:trade_price_mud
			}
			set_variable = { 
				name = info_trade_county_price_fuel
				value = county.var:trade_price_fuel
			}
			set_variable = { 
				name = info_trade_county_price_fiber
				value = county.var:trade_price_fiber
			}
			set_variable = { 
				name = info_trade_county_price_timber
				value = county.var:trade_price_timber
			}
			set_variable = { 
				name = info_trade_county_price_metal
				value = county.var:trade_price_metal
			}
			set_variable = { 
				name = info_trade_county_price_exotics_w
				value = county.var:trade_price_exotics_w
			}
			set_variable = { 
				name = info_trade_county_price_exotics_e
