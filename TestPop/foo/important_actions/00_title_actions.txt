action_can_create_title = {
	combine_into_one = yes
	check_create_action = {
		if = {
			limit = {
				is_character = yes
			}
			ordered_alert_creatable_title = {
				order_by = tier
				max = 99
				check_range_bounds = no
				try_create_important_action = {
					important_action_type = action_can_create_title
					landed_title = this
				}
			}
		}
	}
	effect = {
		scope:landed_title = {
			open_view_data = {
				view = title_view_window
			}
		}
	}
}
